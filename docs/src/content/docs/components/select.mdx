Selects allow you to choose items from a menu of predefined options.

```demo:open
<pc-select>
    <pc-option value="option-1">Option 1</pc-option>
    <pc-option value="option-2">Option 2</pc-option>
    <pc-option value="option-3">Option 3</pc-option>
    <pc-option value="option-4">Option 4</pc-option>
    <pc-option value="option-5">Option 5</pc-option>
    <pc-option value="option-6">Option 6</pc-option>
</pc-select>
```

<pc-callout appearance="primary">
    <pc-icon

## Demos

### Labels

Use the `label` attribute to give the select an accessible label. For labels that contain HTML, use the `label` slot instead.

```demo
<pc-select label="Select an option">
    <pc-option value="option-1">Option 1</pc-option>
    <pc-option value="option-2">Option 2</pc-option>
    <pc-option value="option-3">Option 3</pc-option>
</pc-select>
```

### Hints

Add a descriptive hint to a select with the `hint` attribute. For hints that contain HTML, use the `hint` slot instead.

```demo
<pc-select label="Experience" hint="Please tell us your skill level.">
    <pc-option value="novice">Novice</pc-option>
    <pc-option value="intermediate">Intermediate</pc-option>
    <pc-option value="advanced">Advanced</pc-option>
</pc-select>
```

### Clearable

Use the `clearable` attribute to make the select clearable. The clear button only appears when an option is selected.

```demo

    <pc-option value="option-2">Option 2</pc-option>
    <pc-option value="option-3">Option 3</pc-option>
</pc-select>
```

### Multiple options

To allow multiple options to be selected, use the `multiple` attribute. It’s a good practice to use the `clearable` attribute as well when this option is enabled.

```demo
<pc-select label="Select a few options" multiple clearable>
    <pc-option value="option-1" selected>Option 1</pc-option>
    <pc-option value="option-2" selected>Option 2</pc-option>
    <pc-option value="option-3" selected>Option 3</pc-option>
    <pc-option value="option-4">Option 4</pc-option>
    <pc-option value="option-5">Option 5</pc-option>
    <pc-option value="option-6">Option 6</pc-option>
</pc-select>
```

### Custom tags

When multiple options can be selected, you can provide custom tags by passing a function to the `getTag` property. Your function can return a string of HTML, a [Lit template](https://lit.dev/docs/templates/overview) or an [`HTMLElement`](https://developer.mozilla.org/en-US/docs/Web/API/HTMLElement). The `getTag()` function will be called for each option. The first argument is a `<pc-option>` element and the second argument is the tag’s index (its position in the tag list).

Remember that custom tags are rendered in a shadow root. To style them, you can use the `style` attribute in your template or you can add your own [parts](/docs/get-started/customisation/#css-parts) and target them with the [`::part()`](https://developer.mozilla.org/en-US/docs/Web/CSS/::part) selector.

```demo
<pc-select
    class="custom-tags"
    placeholder="Select one"
    multiple
    clearable
>
    <pc-option value="e-mail" selected>
        <pc-icon
            library="default"
            icon-style="solid"
            name="envelope"
            slot="prefix"
        ></pc-icon>
        E‐mail
    </pc-option>
    <pc-option value="phone" selected>
        <pc-icon
            library="default"
            icon-style="solid"
            name="phone"
            slot="prefix"
        ></pc-icon>
        Phone
    </pc-option>
    <pc-option value="chat">
        <pc-icon
            library="default"
            icon-style="solid"
            name="message"
            slot="prefix"
        ></pc-icon>
        Chat
    </pc-option>
</pc-select>

<script>
    const select = document.querySelector(".custom-tags");

    select.getTag = (option, index) => {
        const name = option.querySelector('pc-icon[slot="prefix"]').name;

        // A string, Lit template or HTMLElement can be returned here
        return `
            <pc-tag data-value="${option.value}" removable>
                <pc-icon
                    library="default"
                    icon-style="solid"
                    name="${name}"
                    style="padding-inline-end: var(--pc-spacing-s)"
                ></pc-icon>
                ${option.label}
            </pc-tag>
        `;
    };
</script>
```

### Lazy loading options

Lazy loading options is very hard to get right. `<pc-select>` largely follows how a native `<select>` works.

Here are the following conditions:

- If a `<pc-select>` is created without any options, but is given a `value` attribute, its `value` will be `""`, and then when options are added, if any of the options have a value equal to the `<pc-select>` value, the value of the `<pc-select>` will be equal that of the option.

For example, `<pc-select value="foo">` will have a value of `""` until `<pc-option value="foo">Foo</pc-option>` connects, at which point its value will become `"foo"` when submitting.

- If a `<pc-select multiple>` with an initial value has multiple values, but only some of the options are present, it will only respect the options that are present, and if a selected option is loaded in later, **and** the value of the select has not changed via user interaction or direct property assignment, it will add the selected option to the form value and to the `value` property of the select.

This can be hard to conceptualise, so here’s a fairly large example showing how lazy‐loaded options work with `<pc-select>` and `<pc-select multiple>` when given initial value attributes. Feel free to play around with it in this code demo.

```demo
<form id="lazy-options-example">
    <div>
        <pc-select name="select-1" label="Single select (with existing options)">
            <pc-option value="bar">Bar</pc-option>
            <pc-option value="baz">Baz</pc-option>
        </pc-select>
        <br />
        <pc-button type="button">Add “foo” option</pc-button>
    </div>

    <br />

    <div>
        <pc-select
            label="Single select (with no existing options)"
            name="select-2"
        ></pc-select>
        <br />
        <pc-button type="button">Add “foo” option</pc-button>
    </div>

    <br />

    <div>
        <pc-select
            label="Multiple select (with existing options)"
            name="select-3"
            multiple
        >
            <pc-option value="bar" selected>Bar</pc-option>
            <pc-option value="baz" selected>Baz</pc-option>
        </pc-select>
        <br />
        <pc-button type="button">Add “foo” option</pc-button>
    </div>

    <br />

    <div>
        <pc-select
            label="Multiple select (with no existing options)"
            name="select-4"
            multiple
        ></pc-select>
        <br />
        <pc-button type="button">Add “foo” option</pc-button>
    </div>

    <br />

    <div style="display: flex; gap: var(--pc-spacing-s)">
        <pc-button appearance="primary" type="submit">Show FormData</pc-button>
        <pc-button type="reset">Reset</pc-button>
    </div>

    <br />

    <pre hidden><code id="lazy-options-example-form-data"></code></pre>
</form>

<script>
    function addFooOption(event) {
        const addFooButton = event.target.closest(
            'pc-button[type="button"]'
        );

        if (!addFooButton) {
            return;
        }

        const select = addFooButton.parentElement.querySelector("pc-select");

        if (select.querySelector('pc-option[value="foo"]')) {
            return;
        }

        const option = document.createElement("pc-option");

        option.setAttribute("value", "foo");
        option.setAttribute("selected", "");
        option.innerText = "Foo";

        select.append(option);
    }

    function handleLazySubmit(event) {
        event.preventDefault();

        const formData = new FormData(event.target);
        const codeElement = document.querySelector("#lazy-options-example-form-data");
        const obj = {};

        for (const key of formData.keys()) {
            const value = formData.getAll(key).length > 1 ? formData.getAll(key) : formData.get(key);
            obj[key] = value;
        }

        codeElement.textContent = JSON.stringify(obj, null, 4);
        const preElement = codeElement.parentElement;
        preElement.removeAttribute("hidden");
    }

    const container = document.querySelector("#lazy-options-example");

    container.addEventListener("click", addFooOption);
    container.addEventListener("submit", handleLazySubmit);
</script>
```
